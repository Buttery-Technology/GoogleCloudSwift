  import Testing
  @testable import GoogleCloudSwift

  /// Integration tests against real Google Cloud APIs
  /// Run with: swift test --filter IntegrationTests
  /// Requires GOOGLE_APPLICATION_CREDENTIALS environment variable
  @Suite("Integration Tests", .disabled("Enable manually for real API testing"))
  struct IntegrationTests {

      let projectId: String
      let credentials: GoogleCloudServiceAccountCredentials

      init() throws {
          // Load from environment or direct path
          guard let credPath =
  ProcessInfo.processInfo.environment["GOOGLE_APPLICATION_CREDENTIALS"] else {
              throw TestError("GOOGLE_APPLICATION_CREDENTIALS not set")
          }

          self.credentials = try GoogleCloudServiceAccountCredentials.load(from:
   credPath)
          self.projectId = credentials.projectId
      }

      // MARK: - Authentication Tests

      @Test("Authenticate and get access token")
      func testAuthentication() async throws {
          let authClient = GoogleCloudAuthClient(
              credentials: credentials,
              scopes: ["https://www.googleapis.com/auth/cloud-platform"]
          )

          let token = try await authClient.getAccessToken()
          #expect(!token.accessToken.isEmpty)
          #expect(!token.isExpired)
      }

      // MARK: - Compute Engine Tests

      @Test("List Compute Engine zones")
      func testListZones() async throws {
          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let computeAPI = GoogleCloudComputeAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              let zones = try await computeAPI.listZones()
              #expect(!zones.items.isEmpty)
              print("Found \(zones.items.count) zones")
          }
      }

      @Test("List Compute Engine instances")
      func testListInstances() async throws {
          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let computeAPI = GoogleCloudComputeAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              // List instances in us-central1-a (adjust zone as needed)
              let instances = try await computeAPI.listInstances(zone:
  "us-central1-a")
              print("Found \(instances.items.count) instances in us-central1-a")
          }
      }

      // MARK: - Cloud Storage Tests

      @Test("List Storage buckets")
      func testListBuckets() async throws {
          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let storageAPI = GoogleCloudStorageAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              let buckets = try await storageAPI.listBuckets()
              print("Found \(buckets.items.count) buckets")
          }
      }

      @Test("Create, upload, download, and delete object")
      func testStorageOperations() async throws {
          let testBucket = "googlecloud-swift-test-\(projectId)"
          let testObject = "test-object-\(UUID().uuidString).txt"
          let testContent = "Hello from GoogleCloudSwift!"

          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let storageAPI = GoogleCloudStorageAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              // Create bucket (may already exist)
              do {
                  _ = try await storageAPI.createBucket(
                      name: testBucket,
                      location: "US"
                  )
                  print("Created bucket: \(testBucket)")
              } catch {
                  print("Bucket may already exist: \(error)")
              }

              // Upload object
              let uploadData = Data(testContent.utf8)
              _ = try await storageAPI.uploadObject(
                  bucket: testBucket,
                  name: testObject,
                  data: uploadData,
                  contentType: "text/plain"
              )
              print("Uploaded object: \(testObject)")

              // Download object
              let downloadedData = try await storageAPI.downloadObject(
                  bucket: testBucket,
                  object: testObject
              )
              let downloadedContent = String(data: downloadedData, encoding:
  .utf8)
              #expect(downloadedContent == testContent)
              print("Downloaded and verified object content")

              // Delete object
              try await storageAPI.deleteObject(bucket: testBucket, object:
  testObject)
              print("Deleted object: \(testObject)")
          }
      }

      // MARK: - Secret Manager Tests

      @Test("Create, access, and delete secret")
      func testSecretManagerOperations() async throws {
          let secretId = "test-secret-\(UUID().uuidString.prefix(8))"
          let secretValue = "super-secret-value"

          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let secretManager = GoogleCloudSecretManagerAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              // Create secret
              let secret = try await secretManager.createSecret(secretId:
  secretId)
              print("Created secret: \(secret.name)")

              // Add secret version
              let version = try await secretManager.addSecretVersion(
                  secretId: secretId,
                  payload: Data(secretValue.utf8)
              )
              print("Added version: \(version.name)")

              // Access secret
              let accessedData = try await secretManager.accessSecretVersion(
                  secretId: secretId,
                  version: "latest"
              )
              let accessedValue = String(data: accessedData, encoding: .utf8)
              #expect(accessedValue == secretValue)
              print("Accessed and verified secret value")

              // Delete secret
              try await secretManager.deleteSecret(secretId: secretId)
              print("Deleted secret: \(secretId)")
          }
      }

      // MARK: - Cloud Run Tests

      @Test("List Cloud Run services")
      func testListCloudRunServices() async throws {
          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let runAPI = GoogleCloudRunAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

      }

      @Test("List service accounts")

      }

      @Test("List log entries")
      func testListLogEntries() async throws {
          try await GoogleCloudHTTPClientFactory.withHTTPClient { httpClient in
              let loggingAPI = GoogleCloudLoggingAPI(
                  projectId: projectId,
                  credentials: credentials,
                  httpClient: httpClient
              )

              let entries = try await loggingAPI.listLogEntries(
                  filter: "resource.type=\"global\"",
                  pageSize: 10
              )
              print("Found \(entries.entries?.count ?? 0) log entries")
          }
      }
  }

      let message: String
      init(_ message: String) { self.message = message }
      var description: String { message }
  }

  Step 1: Create a Service Account

  Option A: Google Cloud Console (Website)

  1. Go to https://console.cloud.google.com
  2. Select your project from the dropdown at the top
  3. Navigate to IAM & Admin → Service Accounts (or search "Service Accounts")
  4. Click + Create Service Account
  5. Fill in:
    - Service account name: googlecloud-swift-test
    - Service account ID: auto-fills
    - Description: Integration testing for GoogleCloudSwift
  6. Click Create and Continue
  7. Grant roles (click + Add Another Role for each):
    - Compute Viewer (for Compute Engine read access)
    - Storage Admin (for Cloud Storage)
    - Secret Manager Admin (for Secret Manager)
    - Cloud Run Viewer (for Cloud Run)
    - Service Account Viewer (for IAM)
    - Logs Viewer (for Cloud Logging)
  8. Click Continue → Done

  Option B: gcloud CLI

  export PROJECT_ID="your-project-id"

      --display-name="GoogleCloudSwift Integration Tests"
